Examples

maf_three comes with pre made examples. To list the available examples, call:

python3 -m maf_three.examples list

To run a specific example call:

python3 -m maf_three.examples example_name

projector

 1# Projector
 2
 3from maf_three.scanner import Scanner
 4from maf_three.Settings.projector import Image, Orientation, Pattern, Projector, Rect, Source
 5from maf_three.Settings.video import Format
 6from maf_three.V3Task import V3Task
 7
 8import time
 9import numpy as np
10
11
12def main():
13
14    try:
15        scanner = Scanner(OnTask=None, OnMessage=None, OnBuffer=None)
16        scanner.Connect("ws://matterandform.local:8081")
17
18        #### Turn ON
19        print('Turn ON')
20        scanner.SendTask(0, V3Task.SetProjector, Projector(on=True, brightness=1.0, color=[1,1,1]))
21        time.sleep(1)
22
23        #### Project Red
24        print('Project Red')
25        scanner.SendTask(1, V3Task.SetProjector, Projector(color=[1,0,0]))
26        time.sleep(1)
27
28        #### Project Green
29        print('Project Green')
30        scanner.SendTask(2, V3Task.SetProjector, Projector(color=[0,1,0]))
31        time.sleep(1)
32
33        #### Project Blue
34        print('Project Blue')
35        scanner.SendTask(3, V3Task.SetProjector, Projector(color=[0,0,1]))
36        time.sleep(1)
37
38        #### Project Vertical Pattern
39        print('Project Vertical Pattern')
40        scanner.SendTask(4, V3Task.SetProjector, Projector(pattern=Pattern(Orientation.Vertical,frequency=4,phase=1)))
41        time.sleep(1)
42
43        #### Project Horizontal Pattern
44        print('Project Horizontal Pattern')
45        scanner.SendTask(5, V3Task.SetProjector, Projector(pattern=Pattern(Orientation.Horizontal,frequency=4,phase=1)))
46        time.sleep(1)
47
48        #### Project an image
49        print('Project Image')
50        width = 640
51        height = 480
52        image = np.zeros([height, width, 3], np.uint8)
53        for y in range(height):
54            for x in range(0, width):
55                image[y,x] = (
56                    255 * y / height , # Blue
57                    255 * x / width , # Green
58                    255 - 255 * y / height # Red
59                )
60        source = Source(format = Format.BGR888, width=width, height=height)
61        scanner.SendTaskWithBuffer(6, V3Task.SetProjector, image.tobytes(),Projector(image=Image(source=source, target=Rect(100,100,640, 480))) )
62        time.sleep(1)
63
64        #### Turn OFF
65        print('Turn OFF')
66        scanner.SendTask(7, V3Task.SetProjector, Projector(on=False))
67
68    except Exception as error:
69        print('Error : ', error)
70    except:
71        print('Error')
72
73    finally: 
74        if scanner.IsConnected():
75            scanner.Disconnect()
76
77if __name__ == "__main__":
78    main()

turntable

 1# Turntable
 2
 3from maf_three.scanner import Scanner
 4from maf_three.task import Task, TaskState
 5from maf_three.V3Task import V3Task
 6
 7import time
 8
 9def main():
10
11    hasTurntable = None
12    rotating = False
13
14    def OnTask(task:Task):
15        global hasTurntable, rotating
16
17        # HasTurntable
18        if task.Type == V3Task.HasTurntable and task.State == TaskState.Completed:
19            hasTurntable = task.Output
20
21        # RotateTurntable
22        elif task.Type == V3Task.RotateTurntable:
23            if task.State == TaskState.Started:
24                rotating = True
25            elif task.State == TaskState.Completed:
26                rotating = False
27
28    try:
29        scanner = Scanner(OnTask=OnTask, OnMessage=None, OnBuffer=None)
30        scanner.Connect("ws://matterandform.local:8081")
31
32        # Check if a Turntable is connected to the scanner
33        scanner.SendTask(0, V3Task.HasTurntable)
34        while hasTurntable == None:
35            pass
36        if hasTurntable == False:
37            raise Exception('There is no turntable connected to the scanner')
38
39        while True:
40
41            while rotating:
42                time.sleep(0.1)
43
44            userInput = input('Enter an angle or "q" to quit\n')
45
46            if userInput.lower() == 'q':
47                break
48
49            try:
50                angle = float(userInput)
51                rotating = True
52                scanner.SendTask(0, V3Task.RotateTurntable, angle)
53            except:
54                pass
55
56
57    except Exception as error:
58        print('Error : ', error)
59    except:
60        print('Error')
61
62    finally: 
63        if scanner.IsConnected():
64            scanner.Disconnect()
65
66if __name__ == "__main__":
67    main()

task

 1# Task
 2import time
 3
 4from maf_three.scanner import Scanner
 5from maf_three.Settings.camera import Camera
 6from maf_three.Settings.capture import Capture
 7from maf_three.Settings.projector import Projector
 8from maf_three.Settings.scan import Scan
 9
10from maf_three.task import Task, TaskState
11from maf_three.V3Task import V3Task
12
13done = False
14
15def main():
16
17    def OnTask(task:Task):
18        global done
19
20        # Inspect Task State
21        match task.State:
22
23            # Started
24            case TaskState.Started:
25                print('\nTask Started:\tIndex:', task.Index, ' - ' ,task.Type)
26
27            # Completed
28            case TaskState.Completed:
29                print('Task Completed:\tIndex:', task.Index, ' - ' , task.Type)
30                if(task.Index == 1):
31                    done = True
32            
33            # Failed
34            case TaskState.Failed:
35                print('Task Failed:\tIndex:', task.Index, ' - ', task.Type, ' - Error:', task.Error)
36                if(task.Index == 1):
37                    done = True
38
39        # Track Task Progress
40        if task.Progress != None:
41            progress = task.Progress['ScanProcess']
42            print(progress['current'] , '/', progress['total'], '-', progress['step'])
43
44
45    try:
46        # Connect
47        scanner = Scanner(OnTask=OnTask, OnMessage=None, OnBuffer=None)
48        scanner.Connect("ws://matterandform.local:8081")
49
50        # Try to scan without input => Will trigger an error
51        scanner.SendTask(0, V3Task.NewTestScan)
52
53        # Scan
54        scan = Scan(
55            Camera(analogGain=256,digitalGain=256,exposure=50000),
56            Capture(), 
57            Projector(brightness=0.5)
58        )
59        scanner.SendTask(1, V3Task.NewTestScan, scan)
60
61        # Wait for the tasks to finish
62        while not done:
63            time.sleep(0.1)
64
65
66    except Exception as error:
67        print('Error : ', error)
68    except:
69        print('Error')
70
71    finally: 
72        if scanner.IsConnected():
73            scanner.Disconnect()
74
75if __name__ == "__main__":
76    main()
